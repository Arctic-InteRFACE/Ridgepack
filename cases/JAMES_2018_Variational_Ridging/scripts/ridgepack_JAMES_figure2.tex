% a plane with varying y coordinate
\newcommand{\planemaina}[3]{
	(-2, #1, 2) --
	++(8, 0.3, 0) --
	++(0, -5, -8) --
	++(-8, -0.3, 0) --
	cycle;
	\node[fill=white,inner sep=2pt, anchor=east] at (-2, #1, 2) {#2};
	\node[anchor=north, black!40] at (5.8, #1+0.2, 5.5) {#3};
	}
